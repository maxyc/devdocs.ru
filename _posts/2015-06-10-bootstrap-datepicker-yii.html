Реализует все функции нормального дейтпикера. ЗДЕСЬ можно поиграться

Кидаем содержимое архива в components
Чтобы получилось такое


Использование



       $this->widget('application.components.was.WasDatepicker',array(

           'model'=>$model,

           'attribute'=>'create_time',

       //model + attribute or 'name'=>'nameInput',

           'options'=>array(

               'language'=>'ru',

               'format'=>'yyyy-mm',

               'autoclose'=>'true',

               'minViewMode'=>1,

               'keyboardNavigation'=>true

            ),

            'htmlOptions'=>array(

               'value'=>date("Y-m"),

             ),

      ));
