пруф
1. качаем дополнение
кидаем папку yii-debug-toolbar в extensions
2.в configmain
не забудьте про

'preload' => array('log'),

     'log' => array(

            'class' => 'CLogRouter',

            'enabled' => true,

            'routes' => array(

                #...

                array(

                    'class' => 'CFileLogRoute',

                    'levels' => 'error, warning',

                ),

                array(

                    'class' => 'application.extensions.yii-debug-toolbar.YiiDebugToolbarRoute',

                    'ipFilters' => array('127.0.0.1', '*.*.*.*'),

                ),

            ),

        ),

    ),




в ipfilters прописываем свой айпишник
3. в поключении к базе добавляем
-- 
 'enableProfiling'=>true, 
--

т.е. должно получиться следующее
--



return array(

          'class' => 'DBConnection',

            'connectionString' => 'pgsql:host=*.ru;port=5432;dbname=*',

            'emulatePrepare' => true,

            'username' => '*',

            'password' => '*',

            'charset' => 'utf8',

            'defaultSchema' => '*',

            'enableProfiling'=>true,

);


--

при наведении в правый верхний угол появится плюсик. нажимаем его и вуаля
